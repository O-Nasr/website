1. Exception Hierarchy

System.Exception                      ← base of everything
    ├── System.SystemException         ← thrown by .NET runtime
    │       ├── NullReferenceException
    │       ├── IndexOutOfRangeException
    │       ├── InvalidCastException
    │       ├── StackOverflowException
    │       └── OutOfMemoryException
    │
    └── System.ApplicationException   ← thrown by application code
            └── YourCustomException   ← your own exceptions

2. Catching Specific Exceptions — Most Specific First

public void ProcessOrder(string orderId)
{
    try
    {
        var order = GetOrder(orderId);
        SaveOrder(order);
    }
    catch (OrderNotFoundException ex)  // most specific first
    {
        Console.WriteLine("Order not found: " + ex.OrderId);
    }
    catch (DatabaseException ex)       // more general
    {
        Console.WriteLine("DB error: " + ex.Message);
    }
    catch (Exception ex)               // catch-all — last resort
    {
        Console.WriteLine("Unexpected: " + ex.Message);
        throw; // rethrow after logging
    }
    finally
    {
        // ALWAYS runs — exception, return, or normal flow
        Console.WriteLine("Processing complete");
    }
}

3. throw vs throw ex — Most Important Distinction

⚠️ Important: throw preserves the original stack trace. throw ex RESETS the stack trace — you lose where the error originally happened. Always use throw.

// ❌ throw ex — destroys evidence
catch (Exception ex)
{
    throw ex; // stack trace now starts HERE — lost original location
}

// ✅ throw — preserves full history
catch (Exception ex)
{
    Logger.Log(ex);
    throw; // original stack trace fully preserved
}

// Stack trace with throw ex:
// at MethodB()  ← starts here — MethodC info LOST 😱
// at MethodA()

// Stack trace with throw:
// at MethodC() line 5  ← original source preserved ✅
// at MethodB()
// at MethodA()

4. Finally Block

Finally ALWAYS runs — even with return statements, exceptions, or break statements inside try.
public string Test()
{
    try
    {
        return "from try"; // return is here
    }
    finally
    {
        Console.WriteLine("finally runs!"); // STILL runs before returning
    }
}
// Output: "finally runs!" then returns "from try"

// finally only skipped on:
// Environment.FailFast() or power failure

5. Custom Exceptions

public class OrderNotFoundException : Exception
{
    public string OrderId { get; }
    public DateTime AttemptedAt { get; }

    public OrderNotFoundException(string orderId)
        : base($"Order '{orderId}' was not found")
    {
        OrderId = orderId;
        AttemptedAt = DateTime.UtcNow;
    }

    // Always include InnerException constructor
    public OrderNotFoundException(string orderId, Exception innerException)
        : base($"Order '{orderId}' was not found", innerException)
    {
        OrderId = orderId;
        AttemptedAt = DateTime.UtcNow;
    }
}

// Throw it
throw new OrderNotFoundException(orderId);

// Catch it specifically
catch (OrderNotFoundException ex)
{
    Console.WriteLine($"Order {ex.OrderId} not found at {ex.AttemptedAt}");
}

6. Global Exception Handler — ASP.NET Core

public class GlobalExceptionMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger _logger;

    public GlobalExceptionMiddleware(RequestDelegate next, ILogger logger)
    {
        _next = next;
        _logger = logger;
    }

    public async Task Invoke(HttpContext context)
    {
        try
        {
            await _next(context);
        }
        catch (OrderNotFoundException ex)
        {
            _logger.LogWarning(ex, "Order not found");
            await HandleException(context, StatusCodes.Status404NotFound, ex.Message);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Unexpected error");
            await HandleException(context, StatusCodes.Status500InternalServerError, "An unexpected error occurred");
        }
    }

    private async Task HandleException(HttpContext ctx, int statusCode, string message)
    {
        ctx.Response.StatusCode = statusCode;
        ctx.Response.ContentType = "application/json";
        await ctx.Response.WriteAsJsonAsync(new { error = message, statusCode });
    }
}

// Register in Program.cs
app.UseMiddleware<GlobalExceptionMiddleware>();

7. Exception Best Practices

Never Swallow Exceptions Silently

// ❌ Empty catch — exception disappears silently
catch (Exception ex) { }

// ✅ Always log at minimum
catch (Exception ex)
{
    _logger.LogError(ex, "Something went wrong");
    throw;
}

Don't Use Exceptions for Flow Control

// ❌ Using exception for normal logic
public bool UserExists(string email)
{
    try { var user = GetUser(email); return true; }
    catch (UserNotFoundException) { return false; } // expensive!
}

// ✅ Simple check
public bool UserExists(string email)
{
    return _repository.Exists(email);
}

Always Include InnerException When Wrapping

// ❌ Loses original exception
catch (SqlException ex)
{
    throw new DatabaseException("DB operation failed");
}

// ✅ Preserves original as InnerException
catch (SqlException ex)
{
    throw new DatabaseException("DB operation failed", ex);
    //                                                  ↑ original preserved

9. Interview Questions

Q: What is the difference between throw and throw ex?

🎯 Interview Answer: throw preserves the original stack trace showing exactly where the exception first occurred. throw ex resets the stack trace making it appear the exception originated at the catch block — you lose the original error location. Always use throw when rethrowing. Only use throw ex inside a new exception constructor to wrap it, passing the original as InnerException.

Q: When does the finally block run?

🎯 Interview Answer: The finally block always runs regardless of whether an exception was thrown, whether a return statement was hit inside try, or whether the code completed normally. The only scenarios where finally does not run are Environment.FailFast() which terminates the process immediately, or a machine power failure. Finally is used for cleanup code like disposing connections and closing files.

Q: When should you catch an exception vs let it bubble up?

🎯 Interview Answer: Catch an exception when you can actually recover from it, need to translate it to a different exception type, need to log it at this specific layer, or need to return a specific HTTP response to the client. Let it bubble up when you cannot do anything useful with it, a higher layer handles it better, or the global exception handler will catch it anyway. Never catch exceptions just to rethrow them without adding value.
